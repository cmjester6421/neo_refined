\section*{Pre-Installation Inspection}

\begin{itemize}
\item[$\square$] Inspect truck bed underside for crossmembers and structural integrity
\item[$\square$] Identify all utility lines (fuel, brake, electrical) under bed
\item[$\square$] Photograph crossmember locations and spacing before installation
\item[$\square$] Check for rust or corrosion on bed frame
\item[$\square$] Verify truck battery is in good condition (12V, 500+ CCA)
\item[$\square$] Test all existing truck electrical systems operational
\item[$\square$] Confirm cap is properly secured to truck bed
\item[$\square$] Measure available mounting space in truck bed
\end{itemize}

\section*{Mounting Installation}

\subsection*{Structural Preparation}

\begin{itemize}
\item[$\square$] Obtain 1/4" steel support plate (minimum 12" x 12")
\item[$\square$] Verify plate spans at least 2 crossmembers
\item[$\square$] Locate rear crossmember (tapered channel) near tailgate
\item[$\square$] Locate front crossmember near cab area
\item[$\square$] Mark mounting hole locations on support plate
\item[$\square$] Drill 3/8" holes in support plate for base bolts
\item[$\square$] Test fit support plate on crossmembers
\item[$\square$] Verify plate is level using spirit level
\end{itemize}

\subsection*{Bolting and Fastening}

\begin{itemize}
\item[$\square$] Obtain Grade 5 hardened steel bolts (identified by 3 hash marks)
\item[$\square$] Use 3/8" x 1-1/2" bolts minimum for mounting
\item[$\square$] Obtain external tooth lock washers for all fasteners
\item[$\square$] Obtain fender washers (spreads load wider)
\item[$\square$] Drill pilot holes through bed floor using 5/16" bit
\item[$\square$] Drill through holes in crossmembers (countersink if possible)
\item[$\square$] Insert bolts from top of bed floor
\item[$\square$] Apply Grade 8 threadlocker (medium strength) to bolt threads
\item[$\square$] Hand-tighten all bolts first
\item[$\square$] Torque bolts to 60-80 ft-lbs in alternating pattern (like wheel pattern)
\item[$\square$] Verify no movement after torquing
\item[$\square$] Allow threadlocker to cure per manufacturer specs (typically 24 hours)
\item[$\square$] Re-check bolt torque after 24-hour cure period
\end{itemize}

\subsection*{Sealing and Protection}

\begin{itemize}
\item[$\square$] Apply silicone sealer around all bed floor bolt penetrations
\item[$\square$] Apply underbody coating to bolt ends and fasteners on underside
\item[$\square$] Inspect bed floor for new cracks or deformation
\item[$\square$] Install rubber grommets around any exposed wiring holes
\item[$\square$] Allow all sealants to cure per manufacturer specifications
\end{itemize}

\subsection*{Hoist Base Mounting}

\begin{itemize}
\item[$\square$] Verify support plate is secure before proceeding
\item[$\square$] Position Bruno lift base on support plate
\item[$\square$] Ensure base is centered and level
\item[$\square$] Drill base mounting holes (typically 4-6 holes)
\item[$\square$] Use 3/8" x 1" Grade 5 bolts for base attachment
\item[$\square$] Apply lock washers and fender washers to base bolts
\item[$\square$] Torque base bolts to 40-50 ft-lbs
\item[$\square$] Verify lift arm moves freely through full range of motion
\item[$\square$] Check for any binding or misalignment
\end{itemize}

\section*{Electrical Installation}

\subsection*{Battery Connection}

\begin{itemize}
\item[$\square$] Disconnect negative terminal from vehicle battery
\item[$\square$] Obtain 10 AWG (or 8 AWG for longer runs) power cable
\item[$\square$] Measure distance from battery to control box location
\item[$\square$] Install 30-amp ATO fuse holder at battery positive terminal
\item[$\square$] Install 30-amp fuse in fuse holder
\item[$\square$] Crimp ring terminal to red power cable at battery
\item[$\square$] Connect ring terminal to fuse holder positive side
\item[$\square$] Route black ground cable from battery negative to chassis ground
\item[$\square$] Reconnect battery negative terminal
\item[$\square$] Test fuse holder for secure connection (no movement)
\end{itemize}

\subsection*{Interior Power Routing}

\begin{itemize}
\item[$\square$] Route red power cable through firewall using existing grommet
\item[$\square$] If new hole required: drill 3/16" hole in firewall
\item[$\square$] Install rubber grommet in firewall hole
\item[$\square$] Apply silicone sealer around grommet
\item[$\square$] Route power cable along driver's door sill (under carpet if possible)
\item[$\square$] Run cable behind seat or along interior trim
\item[$\square$] Do NOT route under vehicle or near exhaust
\item[$\square$] Secure cable to interior using adhesive-backed wire clips every 12 inches
\item[$\square$] Cable should not contact sharp edges or moving parts
\item[$\square$] Test for any pinch points or abrasion risks
\end{itemize}

\subsection*{Bed Floor Wiring}

\begin{itemize}
\item[$\square$] Drill 11/16" hole in bed floor near lift base (access to motor harness)
\item[$\square$] Install rubber grommet in bed floor hole
\item[$\square$] Apply silicone sealer around grommet from underside
\item[$\square$] Route motor harness through grommet
\item[$\square$] Connect motor harness leads to control box per wiring diagram
\item[$\square$] Use waterproof connectors (solder or crimp, shrink-tube sealed)
\item[$\square$] Do NOT use standard wire connectors (corrosion risk)
\item[$\square$] Ground motor case to vehicle chassis with ring terminal and screw
\end{itemize}

\subsection*{Control Box Installation}

\begin{itemize}
\item[$\square$] Mount control box inside cab (behind driver's seat recommended)
\item[$\square$] Use waterproof plastic enclosure
\item[$\square$] Ensure ventilation holes prevent moisture accumulation
\item[$\square$] Secure control box to seat frame or floor with L-brackets
\item[$\square$] Connect positive power cable to control box input
\item[$\square$] Connect ground cable to control box ground terminal
\item[$\square$] Verify polarity before powering on
\item[$\square$] Connect motor leads to control box motor terminals
\item[$\square$] Test all button functions before final closure
\end{itemize}

\subsection*{Wireless Receiver Installation (If Applicable)}

\begin{itemize}
\item[$\square$] Mount receiver in waterproof enclosure in truck bed
\item[$\square$] Position antenna clear of metal obstructions
\item[$\square$] Route antenna cable along interior of bed cap
\item[$\square$] Connect receiver output to control box relay terminals
\item[$\square$] Program remote per manufacturer instructions
\item[$\square$] Test wireless range (minimum 50 feet)
\item[$\square$] Verify no interference from other vehicles/equipment
\end{itemize}

\section*{System Testing}

\subsection*{Safety Testing}

\begin{itemize}
\item[$\square$] Verify all electrical connections are secure and dry
\item[$\square$] Reconnect battery and check for any sparks or damage
\item[$\square$] Turn on truck ignition without starting engine
\item[$\square$] Listen for any unusual electrical sounds
\item[$\square$] Check that all LEDs or indicators on control box illuminate
\item[$\square$] Test emergency stop button if equipped
\item[$\square$] Verify fuse holder does not overheat
\end{itemize}

\subsection*{Motor Function Testing}

\begin{itemize}
\item[$\square$] Start truck engine
\item[$\square$] Engage parking brake (safety critical)
\item[$\square$] Press extend button on control (not holding full load yet)
\item[$\square$] Listen for motor operation---should be smooth, no grinding
\item[$\square$] Observe lift arm extension (should be smooth and level)
\item[$\square$] Press retract button
\item[$\square$] Observe lift arm retraction
\item[$\square$] Verify arm returns to fully closed position
\item[$\square$] Stop truck engine
\item[$\square$] Test wireless remote (if equipped) at various distances
\item[$\square$] Verify all functions work from both cab control and remote
\end{itemize}

\subsection*{Load Testing}

\begin{itemize}
\item[$\square$] Place test load (sandbags or weight) at 25\% of rated capacity
\item[$\square$] Extend lift with test load
\item[$\square$] Observe for any sagging, tilting, or misalignment
\item[$\square$] Retract test load smoothly
\item[$\square$] Increase test load to 50\% of rated capacity
\item[$\square$] Repeat extension and retraction
\item[$\square$] Observe for any vibration or unusual noise
\item[$\square$] Increase test load to 100\% of rated capacity
\item[$\square$] Perform final extension and retraction under full load
\item[$\square$] Monitor current draw (should not exceed motor rating)
\item[$\square$] Verify all fasteners remain tight after load testing
\item[$\square$] Check for any new leaks or damage
\end{itemize}

\subsection*{Post-Testing Inspection}

\begin{itemize}
\item[$\square$] Re-torque all bolts to specifications
\item[$\square$] Inspect wiring for any pinch points or damage
\item[$\square$] Verify no exposed wiring or corrosion
\item[$\square$] Check for battery drain with ignition off
\item[$\square$] Document any anomalies for troubleshooting
\end{itemize}

\section*{Final Setup and Documentation}

\begin{itemize}
\item[$\square$] Create wiring diagram showing all connections for future reference
\item[$\square$] Document controller button functions clearly
\item[$\square$] Keep all manufacturer manuals accessible (glove box recommended)
\item[$\square$] Take photos of completed installation from multiple angles
\item[$\square$] Label all fuses and breakers in truck fuse panel
\item[$\square$] Test system one final time before declaring complete
\item[$\square$] Record installation date and mileage on this checklist
\end{itemize}

\section*{Maintenance Schedule}

\begin{itemize}
\item[$\square$] \textbf{Monthly:} Visually inspect bolts and fasteners for looseness
\item[$\square$] \textbf{Monthly:} Test extension/retraction without load
\item[$\square$] \textbf{Every 6 months:} Re-torque all bolts to specifications
\item[$\square$] \textbf{Every 6 months:} Inspect wiring for corrosion or damage
\item[$\square$] \textbf{Annually:} Full operational load test
\item[$\square$] \textbf{Annually:} Inspect underside for rust or new cracks
\item[$\square$] \textbf{As needed:} Replace any corroded fasteners with Grade 5 replacements
\end{itemize}

\section*{Troubleshooting Reference}

\begin{tabular}{|p{3cm}|p{5cm}|}
\hline
\textbf{Issue} & \textbf{Solution} \\
\hline
Motor won't operate & Check 30A fuse, verify battery voltage (12V+), test wireless range \\
\hline
Slow extension/retraction & Check motor amperage, verify battery charge, inspect for corrosion \\
\hline
Uneven lifting & Verify mounting bolts are equally tight, check for bent lift arm \\
\hline
Motor runs but lift doesn't move & Check for cable slack, verify mechanical connection, inspect for damage \\
\hline
Wireless remote doesn't work & Test antenna position, check receiver batteries, verify programming \\
\hline
Fuse blows repeatedly & Check for short circuit, inspect wiring for damage, verify motor is not seized \\
\hline
\end{tabular}
